    1. **`src/arcgis_agent/services/task_service.py`** — 任务存储服务。
       ```python
       import sqlite3
       import uuid
       import json
       import threading
       from datetime import datetime
       from pathlib import Path
       from dataclasses import dataclass, field
       from typing import Optional
       from arcgis_agent.config import CONFIG_DIR

       @dataclass
       class Task:
           task_id: str = field(default_factory=lambda: str(uuid.uuid4()))
           tool_name: str = ""
           arguments: dict = field(default_factory=dict)
           status: str = "pending"  # pending, running, completed, failed
           result: Optional[dict] = None
           error: Optional[str] = None
           progress: float = 0.0
           created_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())
           updated_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())

       class TaskStore:
           def __init__(self, db_path: Path | None = None):
               self._lock = threading.Lock()
               if db_path is None:
                   db_path = CONFIG_DIR / "tasks.db"
               db_path.parent.mkdir(parents=True, exist_ok=True)
               self._conn = sqlite3.connect(str(db_path), check_same_thread=False)
               self._conn.row_factory = sqlite3.Row
               self._init_db()

           def _init_db(self):
               self._conn.execute("""
                   CREATE TABLE IF NOT EXISTS tasks (
                       task_id TEXT PRIMARY KEY,
                       tool_name TEXT NOT NULL,
                       arguments TEXT DEFAULT '{}',
                       status TEXT DEFAULT 'pending',
                       result TEXT,
                       error TEXT,
                       progress REAL DEFAULT 0.0,
                       created_at TEXT,
                       updated_at TEXT
                   )
               """)
               self._conn.commit()

           def create(self, tool_name: str, arguments: dict | None = None) -> Task:
               task = Task(tool_name=tool_name, arguments=arguments or {})
               with self._lock:
                   self._conn.execute(
                       "INSERT INTO tasks VALUES (?,?,?,?,?,?,?,?,?)",
                       (task.task_id, task.tool_name, json.dumps(task.arguments),
                        task.status, None, None, task.progress,
                        task.created_at, task.updated_at)
                   )
                   self._conn.commit()
               return task

           def get(self, task_id: str) -> Optional[Task]:
               row = self._conn.execute("SELECT * FROM tasks WHERE task_id=?", (task_id,)).fetchone()
               if row is None:
                   return None
               return Task(
                   task_id=row["task_id"], tool_name=row["tool_name"],
                   arguments=json.loads(row["arguments"]), status=row["status"],
                   result=json.loads(row["result"]) if row["result"] else None,
                   error=row["error"], progress=row["progress"],
                   created_at=row["created_at"], updated_at=row["updated_at"]
               )

           def update(self, task_id: str, **kwargs):
               valid = {"status", "result", "error", "progress", "updated_at"}
               updates = {k: v for k, v in kwargs.items() if k in valid}
               if "updated_at" not in updates:
                   updates["updated_at"] = datetime.utcnow().isoformat()
               set_clause = ", ".join(f"{k}=?" for k in updates)
               values = list(updates.values()) + [task_id]
               with self._lock:
                   self._conn.execute(f"UPDATE tasks SET {set_clause} WHERE task_id=?", values)
                   self._conn.commit()

           def list_recent(self, limit: int = 20) -> list[Task]:
               rows = self._conn.execute(
                   "SELECT * FROM tasks ORDER BY created_at DESC LIMIT ?", (limit,)
               ).fetchall()
               tasks = []
               for row in rows:
                   tasks.append(Task(
                       task_id=row["task_id"], tool_name=row["tool_name"],
                       arguments=json.loads(row["arguments"]), status=row["status"],
                       result=json.loads(row["result"]) if row["result"] else None,
                       error=row["error"], progress=row["progress"],
                       created_at=row["created_at"], updated_at=row["updated_at"]
                   ))
               return tasks
       ```

       ```python
       from fastapi import APIRouter, HTTPException
       from arcgis_agent.services.task_service import TaskStore
       from arcgis_agent.api.schemas.tasks import TaskCreate, TaskResult

       router = APIRouter(prefix="/api/v1/tasks", tags=["Tasks"])
       _task_store: TaskStore | None = None

       def get_task_store() -> TaskStore:
           global _task_store
           if _task_store is None:
               _task_store = TaskStore()
           return _task_store

       @router.post("", status_code=201)
       async def create_task(body: TaskCreate):
           store = get_task_store()
           task = store.create(body.tool_name, body.arguments)
           return {"task_id": task.task_id, "status": task.status}

       @router.get("/{task_id}")
       async def get_task(task_id: str):
           store = get_task_store()
           task = store.get(task_id)
           if task is None:
               raise HTTPException(status_code=404, detail="Task not found")
           return task

       @router.get("")
       async def list_tasks(limit: int = 20):
           store = get_task_store()
           tasks = store.list_recent(limit)
           return {"tasks": tasks, "count": len(tasks)}
       ```

       ```python
       from fastapi import APIRouter, UploadFile, File, HTTPException
       from pathlib import Path
       import shutil
       import zipfile
       from arcgis_agent.config import WorkspaceConfig

       router = APIRouter(prefix="/api/v1", tags=["Upload"])
       UPLOAD_DIR = Path.home() / ".arcgis-agent" / "uploads"

       @router.post("/upload")
       async def upload_file(file: UploadFile = File(...)):
           UPLOAD_DIR.mkdir(parents=True, exist_ok=True)
           allowed_ext = {".shp", ".shx", ".dbf", ".prj", ".cpg", ".zip", ".gdb"}
           ext = Path(file.filename).suffix.lower()
           if ext not in allowed_ext and not file.filename.endswith(".gdb"):
               raise HTTPException(400, f"Unsupported file type: {ext}")
           
           dest = UPLOAD_DIR / file.filename
           content = await file.read()
           dest.write_bytes(content)
           
           # 如果是 .zip，解压
           if ext == ".zip":
               extract_dir = UPLOAD_DIR / file.filename.replace(".zip", "")
               extract_dir.mkdir(exist_ok=True)
               with zipfile.ZipFile(str(dest), 'r') as zf:
                   zf.extractall(str(extract_dir))
               return {"uploaded": str(dest), "extracted": str(extract_dir)}
           
           # 如果上传了 .gdb 目录，需要特殊处理
           return {"uploaded": str(dest), "filename": file.filename}
       ```
